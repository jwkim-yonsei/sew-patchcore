# sew-patchcore — Python 3.10 dependency set.
#
# Install in a single resolve (`pip install -r requirements.txt`). Installing
# anomalib[core] first and adding pins later leaves an opencv/numpy conflict.
#
# Why these pins (verified in a clean Python 3.10 venv):
#  * anomalib[core]==1.2.0  — released anomalib; the [core] extra pulls
#    torch / lightning / timm / torchmetrics / torchvision (a bare
#    `pip install anomalib==1.2.0` has none of them and fails to import).
#  * numpy<2                — anomalib[core] pulls imgaug==0.4.0, which uses
#    np.sctypes (removed in NumPy 2.0). numpy<2 resolves to 1.26.x and makes
#    a source patch on imgaug unnecessary.
#  * opencv-python==4.11.0.86 — newer opencv (4.13+) requires numpy>=2;
#    this pin keeps the single resolve consistent with numpy<2.
#  * scikit-learn / python-dotenv / requests — `import anomalib.models`
#    imports the whole model zoo (VLM / video / CFA ...), which needs these;
#    the [core] extra does NOT declare them.
#  * timm==1.0.27 — the __AT__ backbone path crashes on some timm versions
#    (`timm.models has no attribute 'registry'`); this code never uses
#    __AT__, but the version is pinned for a reproducible feature extractor.
#  * matplotlib==3.9.2 — anomalib 1.2.0's image visualizer calls
#    `FigureCanvasAgg.tostring_rgb()`, which matplotlib 3.10 removed
#    (`AttributeError: ... has no attribute 'tostring_rgb'` during test).
#    3.9.2 is the last 3.9.x patch release that keeps that API.

anomalib[core]==1.2.0

numpy<2
opencv-python==4.11.0.86

scikit-learn
python-dotenv
requests

timm==1.0.27

# matplotlib 3.10 removed FigureCanvasAgg.tostring_rgb(), which anomalib
# 1.2.0's visualizer still uses — pin to the last 3.9.x.
matplotlib==3.9.2

# torch family — kept compatible with anomalib 1.2.0 / numpy<1.27 / Python 3.10.
torch==2.4.1
torchvision==0.19.1
lightning==2.4.0
torchmetrics==1.3.2
